### 3.2 GREEN — 实现 `mneme/parsers/txt_parser.py`

```python
def extract_txt_bytes(content: bytes) -> list[BlockDraft]:
    encoding = chardet.detect(content)["encoding"] or "utf-8"
    text = content.decode(encoding, errors="replace")
    paragraphs = [p.strip() for p in text.split("\n\n") if p.strip()]
    return [BlockDraft(block_order=i, content_markdown=p, block_type="paragraph")
            for i, p in enumerate(paragraphs)]
```

运行 → 通过

### 4.2 GREEN — 实现 `mneme/parsers/pdf_parser.py`

```python
import fitz

def extract_pdf_bytes(content: bytes) -> list[BlockDraft]:
    doc = fitz.open(stream=content, filetype="pdf")
    blocks = []
    order = 0
    for page in doc:
        text = page.get_text()
        paragraphs = [p.strip() for p in text.split("\n\n") if p.strip()]
        for para in paragraphs:
            blocks.append(BlockDraft(block_order=order, content_markdown=para, block_type="paragraph"))
            order += 1
    doc.close()
    return blocks
```

运行 → 通过

### 6.2 GREEN — 实现 `mneme/parsers/ocr_parser.py`

```python
from paddleocr import PaddleOCR

_ocr_engine = None

def _get_ocr():
    global _ocr_engine
    if _ocr_engine is None:
        _ocr_engine = PaddleOCR(use_angle_cls=True, lang='ch', show_log=False)
    return _ocr_engine

def extract_image_bytes(content: bytes) -> list[BlockDraft]:
    ocr = _get_ocr()
    import numpy as np
    img_array = np.frombuffer(content, np.uint8)
    import cv2
    img = cv2.imdecode(img_array, cv2.IMREAD_COLOR)
    result = ocr.ocr(img, cls=True)
    lines = []
    for line in result[0]:
        text = line[1][0]
        if text.strip():
            lines.append(text.strip())
    merged = "\n\n".join(lines)
    return [BlockDraft(block_order=0, content_markdown=merged, block_type="paragraph")]
```

运行 → 通过

### 7.2 GREEN — 实现 `mneme/parsers/mime_detect.py`

```python
import mimetypes

EXTENSION_MAP = {
    ".pdf": "application/pdf",
    ".docx": "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
    ".doc": "application/msword",
    ".txt": "text/plain",
    ".md": "text/markdown",
    ".csv": "text/csv",
    ".png": "image/png",
    ".jpg": "image/jpeg",
    ".jpeg": "image/jpeg",
    ".webp": "image/webp",
    ".bmp": "image/bmp",
}

def detect_mime_type(filename: str, content: bytes) -> str:
    ext = Path(filename).suffix.lower()
    if ext in EXTENSION_MAP:
        return EXTENSION_MAP[ext]
    # fallback: python-magic
    try:
        import magic
        return magic.from_buffer(content, mime=True)
    except Exception:
        pass
    return "application/octet-stream"
```

运行 → 通过
